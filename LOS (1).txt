local orbActive = {
    Ethereal = false,
    Yellow = false,
    Red = false
}

local function collectOrb(orbType, maxCount)
    orbActive[orbType] = true
    local count = 0

    while orbActive[orbType] and count < maxCount do
        local args = {
            [1] = "collectOrb",
            [2] = orbType .. " Orb",
            [3] = "City"
        }
        game:GetService("ReplicatedStorage").rEvents.orbEvent:FireServer(unpack(args))
        count = count + 1
    end
end
